class Test 
{
    //主函数
    public static void main(String[] args) 
    {
        int num = 12321;//模拟用户输入5位整数 任务二使用
        
        int score = 55; //模拟考试成绩 任务三使用
        
        //实例化对象 
        Test test = new Test();
        
        //任务二调用，判断是否回文数
        test.isPalidrome(num);
        
        //任务三调用，使用if语句判断成绩等级
        test.if_demo(score);
        
        //任务三调用，使用switch语句判断成绩等级
        test.switch_demo(score);
    }
    
    //任务二：判断5位整数是否位回文数
    public void isPalidrome(int num)
    {  
        //整数转字符串判断位数
        String num_str = Integer.toString(num);
      
        if(num_str.length() != 5)
        {
            System.out.println(+ num + "不是一个五位数");
            return;
        }
		
        //判断是否回文
        int a=num%10,b=num/10000,c=num/10%10,d=num/1000%10;
		
        if(a==b && c==d){
            System.out.println(+ num + "是回文数");
        }else{
            System.out.println(+ num + "不是回文数");
        }
    }
   	
   	//任务三，使用if语句判断成绩等级
    public void if_demo(int score)
    {
    	if(score >=90 && score<=100)
        {
    		System.out.println(+ score + "属于优秀");
    	}
                		
    	if(score >=80 && score<=89)
    	{
    		System.out.println(+ score + "属于良好");
    	}
                		
    	if(score >=70 && score<=79)
    	{
    		System.out.println(+ score + "属于中");
    	}
                		
    	if(score >=60 && score<=69)
    	{
    		System.out.println(+ score + "属于及格");
    	}
                		
    	if(score >=0 && score<=59)
    	{
    		System.out.println(+ score + "属于不及格");
    	}
    }
    
    //任务三，使用switch语句判断成绩等级
    public void switch_demo(int score)
    {	
    	//判断是否是有效成绩
    	if (score > 100 || score < 0)
    	{
            System.out.println("输入有误");
            return;
        }
    		
    	switch (score / 10) 
    	{
            case 10:
    		     System.out.println("优秀");
                 break;
            case 9:
                 System.out.println("优秀");
                 break;
            case 8:
                 System.out.println("良好");
                 break;
            case 7:
                 System.out.println("中等");
                 break;
            case 6:
                 System.out.println("及格");
                 break;
            default:
                 System.out.println("不及格");
           	}
    	} 
}
